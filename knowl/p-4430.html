\begin{align*} \vec{u}&=k_1\vec{v}_1+k_2\vec{v}_2\\ \vecddd{1}{2}{6}&=k_1\vecddd{2}{3}{-5}+k_2\vecddd{7}{11}{-9}\\ \vecddd{1}{2}{6}&=\begin{pmatrix}2& 7\\ 3& 11 \\ -5& -9 \end{pmatrix}\vecd{k_1}{k_2} \end{align*}
