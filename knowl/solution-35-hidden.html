\begin{align*} \vec{u}\cdot(\vec{v}+\vec{w}) &=\norm{\vec{u}}\norm{\vec{v}+\vec{w}}\cos(\theta)\\ &= \norm{\vec{u}}\color{green}{\big(\norm{\vec{v}+\vec{w}}\cos(\theta)\big)}\\ &= \norm{\vec{u}}\big(\color{blue}{\norm{\vec{v}}\cos(\alpha)}+\color{red}{\norm{\vec{w}}\cos(\beta)}\big)\\ &= \norm{\vec{u}}\norm{\vec{v}}\cos(\alpha)+\norm{\vec{u}}\norm{\vec{w}}\cos(\beta)\\ &= \vec{u}\cdot\vec{v}+\vec{u}\cdot\vec{w} \end{align*}
