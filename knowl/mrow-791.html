\begin{align*} T_3(\vec{u}+\vec{v})&=T_3\left(\begin{pmatrix}x_1\\ y_1\end{pmatrix}+\begin{pmatrix}x_2\\ y_2\end{pmatrix}\right)\\ &=T_3\begin{pmatrix}x_1+x_2 \\ y_1+y_2\end{pmatrix}\\ &=\begin{pmatrix}2(x_1+x_2)-(y_1+y_2)\\ x_1+x_2\end{pmatrix}\\ &=\begin{pmatrix}2x_1-y_1\\ x_1\end{pmatrix}+\begin{pmatrix}2x_2-y_2\\ x_2\end{pmatrix}\\ &=T_3\begin{pmatrix}x_1\\ y_1\end{pmatrix}+T_3\begin{pmatrix}x_2\\ y_2\end{pmatrix}\\ &=T_3(\vec{u})+T_3(\vec{v}) \end{align*}
