  2. \begin{align*} \vec{u}+(\vec{v}+\vec{w}) & = (u_1,u_2)+\big((v_1,v_2)+(w_1,w_2)\big)\\ & = (u_1,u_2)+\big(v_1+w_1,v_2+w_2\big)\\ & = \big(u_1+(v_1+w_1),u_2+(v_2+w_2)\big)\\ & = \big(u_1+v_1+w_1,u_2+v_2+w_2\big)\\ & = \big((u_1+v_1)+w_1,(u_2+v_2)+w_2\big)\\ & = \big(u_1+v_1,u_2+v_2\big)+(w_1,w_2)\\ & = \big((u_1,u_2)+(v_1,v_2)\big)+(w_1,w_2)\\ & =(\vec{u}+\vec{v})+\vec{w} \end{align*}
  3. \begin{align*} \vec{u}+\vec{0} & =(u_1,u_2)+(0,0)\\ &=(u_1+0,u_2+0)\\ &=(u_1,u_2)\\ &=\vec{u} \end{align*}
  4. \begin{align*} \vec{u}+(-\vec{u}) & =(u_1,u_2)+(-u_1,-u_2)\\ & =\big(u_1+(-u_1),u_2+(-u_2)\big)\\ & =(u_1-u_1,u_2-u_2)\\ & =(0,0)\\ & =\vec{0} \end{align*}
  5. \begin{align*} (rs)\vec{u} & =(rs)(u_1,u_2)\\ & =((rs)u_1,(rs)u_2)\\ & =(rsu_1,rsu_2)\\ & =(r(su_1),r(su_2))\\ & =r(su_1,su_2)\\ & =r\big(s(u_1,u_2)\big)\\ & =r(s\vec{u}) \end{align*}
  6. \begin{align*} r(\vec{u}+\vec{v}) &=r\big((u_1,u_2)+(v_1,v_2)\big)\\ &=r\big(u_1+v_1,u_2+v_2\big)\\ &=\big(r(u_1+v_1),r(u_2+v_2)\big)\\ &=\big(ru_1+rv_1,ru_2+rv_2\big)\\ &=(ru_1,ru_2)+(rv_1,rv_2)\\ &=r(u_1,u_2)+r(v_1,v_2)\\ &=r\vec{u}+r\vec{v} \end{align*}
  7. \begin{align*} (r+s)\vec{u} &=(r+s)(u_1,u_2)\\ &=\big((r+s)u_1,(r+s)u_2\big)\\ &=\big(ru_1+su_1,ru_2+su_2\big)\\ &=(ru_1,ru_2)+(su_1,su_2)\\ &=r(u_1,u_2)+s(u_1,u_2)\\ &=r\vec{u}+s\vec{u} \end{align*}
