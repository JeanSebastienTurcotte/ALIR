\begin{equation*} \det(A)=a_{3,1}C_{3,1}+a_{3,2}C_{3,2}+a_{3,3}C_{3,3}\text{.} \end{equation*}
\begin{align*} \det(A)&=\begin{vmatrix} 1&2& -3\\ -4&5&6\\ 7&-8&9 \end{vmatrix}\\ &=7*27+(-8)*6+9*13\\ &=258 \end{align*}
