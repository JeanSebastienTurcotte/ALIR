\begin{align*} \norm{\vec{w}}^2 &= \norm{\vec{u}}^2+\norm{\vec{v}}^2-2\norm{\vec{u}}\norm{\vec{v}}\cos(\theta)\\ \cos(\theta) &= \frac{\norm{\vec{u}}^2+\norm{\vec{v}}^2-\norm{\vec{w}}^2}{2\norm{\vec{u}}\norm{\vec{v}}} \\ &= \frac{u_1^2+u_2^2+u_3^2+v_1^2+v_2^2+v_3^2-w_1^2-w_2^2-w_3^2}{2\norm{\vec{u}}\norm{\vec{v}}}\\ &= \frac{u_1^2+u_2^2+u_3^2+v_1^2+v_2^2+v_3^2-(u_1-v_1)^2-(u_2-v_2)^2-(u_3-v_3)^2}{2\norm{\vec{u}}\norm{\vec{v}}}\\ &= \frac{u_1^2+u_2^2+u_3^2+v_1^2+v_2^2+v_3^2-(u_1^2-2u_1v_1+v_1^2)-(u_2^2-2u_2v_2+v_2^2)-(u_3^2-2u_3v_3+v_3^2)}{2\norm{\vec{u}}\norm{\vec{v}}}\\ &=\frac{2u_1v_1+2u_2v_2+2u_3v_3}{2\norm{\vec{u}}\norm{\vec{v}}}\\ \cos(\theta)&=\frac{u_1v_1+u_2v_2+u_3v_3}{\norm{\vec{u}}\norm{\vec{v}}} \end{align*}
