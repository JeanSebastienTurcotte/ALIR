\documentclass[tikz,border=5]{standalone}
\usetikzlibrary{matrix}
\begin{document}
\tikz[x=1em, y=1em, inner sep=1em/6]]
  \matrix [left delimiter={(}, right delimiter={)}] {
  \draw (0,0) \foreach \i [count=\j] in {1,2,1,3,1,2,2,1,1}{ 
    \ifnum\j>1-- ++(0,-1)\fi node [above right] (x-\j) {$1$} -- ++(\i,0)
  }
  (x-1 -| x-\j) node [left]  {Z\'eros au-dessus des pivots}
  (x-1 |- x-\j) node [right] {Z\'eros sous les pivots};
\\};
\end{document}
